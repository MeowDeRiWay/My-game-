local ChestWithdraw = {}

local ChestUtils = require(script.Parent:WaitForChild("ChestUtils"))
local ChestOpen = require(script.Parent:WaitForChild("ChestOpen"))

local ctx = {}

function ChestWithdraw.Init(context)
	ctx = context
end

function ChestWithdraw.Run(player, chest, amount)
	if not ChestUtils.IsStorageChest(chest) then return end

	ChestUtils.InitChest(chest)

	amount = tonumber(amount) or 0
	amount = math.floor(amount)

	if amount <= 0 then return end

	local storedItem, storedAmount = ChestUtils.GetChestData(chest)

	if not storedItem or storedAmount <= 0 then
		ctx.ChestRequest:FireClient(player, "Error", "Сундук порожній")
		return
	end

	local withdrawAmount = math.min(amount, storedAmount)

	ctx.AddItem(player, storedItem, withdrawAmount)

	storedAmount -= withdrawAmount

	if storedAmount <= 0 then
		ChestUtils.SetChestData(chest, nil, 0)
	else
		ChestUtils.SetChestData(chest, storedItem, storedAmount)
	end

	ctx.SendInventory(player)
	ChestOpen.Run(player, chest)
end

return ChestWithdraw
